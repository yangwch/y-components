**Usage**

```jsx
import React from 'react';
import { Checkbox } from 'y-components';

export default () => {
  return (
    <div>
      <label>
        <Checkbox /> Checkbox
      </label>
    </div>
  );
};
```

**Radio**
```jsx
import React from 'react';
import { Checkbox } from 'y-components';

export default () => {
  return (
    <div>
      <label>
        <Checkbox type="radio" /> Radio
      </label>
    </div>
  );
};
```
